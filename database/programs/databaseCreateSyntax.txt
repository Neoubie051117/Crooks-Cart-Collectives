CREATE DATABASE IF NOT EXISTS barangay_system;

USE barangay_system;

CREATE TABLE residents (
    residentID INT AUTO_INCREMENT PRIMARY KEY,
    firstName VARCHAR(50) NOT NULL,
    middleName VARCHAR(50),
    lastName VARCHAR(50) NOT NULL,
    birthdate DATE NOT NULL,
    gender ENUM('Male', 'Female') NOT NULL,
    contact VARCHAR(15) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    username VARCHAR(50) NOT NULL UNIQUE,
    passwordHash VARCHAR(255) NOT NULL,
    houseStreet VARCHAR(100) NOT NULL,
    barangay VARCHAR(50) NOT NULL,
    residentType ENUM('Owner', 'Tenant') NOT NULL,
    civilStatus ENUM('Single', 'Married', 'Widowed') DEFAULT NULL,
    nationality ENUM('Filipino', 'Foreigner') NOT NULL,
    
    proofOfResidencyPath VARCHAR(255) NOT NULL, -- Path to image file
    validIDPath VARCHAR(255) NOT NULL,          -- Path to image file

    dateRegistered TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    isVerified BOOLEAN DEFAULT FALSE             -- Optional: for IT to mark verified
);
